func Fibonacci(n int) int {
	if n <= 2 {
		return 1
	}
	sum := 0
	n1 := 0
	n2 := 1
	for i := 1; i < n; i++ {
		sum = n1 + n2
		n1 = n2
		n2 = sum
	}
	return sum
}
